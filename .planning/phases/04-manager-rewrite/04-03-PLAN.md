reproveHeader(deactivatedHash: string): Promise<ReproveHeaderResult>
  - Runs under StorageProvider lock
  - Calls services.getChainTracker()
  - Finds all ProvenTx with blockHash == deactivatedHash
  - For each: calls reproveProven(ptx)
  - Bulk-updates all successfully reproved ProvenTx records in storage
  - Returns { updated: ProvenTx[], unchanged: ProvenTx[], unavailable: ProvenTx[] }

reproveProven(ptx: ProvenTx, noUpdate?: boolean): Promise<ReproveProvenResult>
  - Calls services.getMerklePath(ptx.txid)
  - Validates merkle path against chain tracker
  - Updates: height, index, merklePath, merkleRoot, blockHash
  - If noUpdate is false, persists to storage
  - Returns { updated?: ProvenTx, unchanged: boolean, unavailable: boolean }
